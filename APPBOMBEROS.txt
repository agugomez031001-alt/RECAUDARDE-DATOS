  var NewAccidentScreen = function(props) {
    var navigation = props.navigation;
    var insets = useSafeAreaInsets();
    var s = useNewAccidentState();
    var theme = s.theme;

    var cameraHook = useCamera();
    var takePhoto = cameraHook.takePhoto;
    var photo = cameraHook.photo;

    var locationHook = useLocation();
    var getCurrentLocation = locationHook.getCurrentLocation;
    var requestPermission = locationHook.requestPermission;

    var mutationHook = useMutation('accidents', 'insert');
    var insertAccident = mutationHook.mutate;

    var handleVehicleChange = useCallback(function(idx, field, val) {
      s.setVehicles(function(prev) {
        var updated = prev.map(function(v, i) {
          if (i !== idx) return v;
          var next = Object.assign({}, v);
          next[field] = val;
          return next;
        });
        return updated;
      });
    }, [s.setVehicles]);

    var handleTakePhoto = function() {
      takePhoto().then(function(result) {
        if (result.error) {
          Platform.OS === 'web' ? window.alert('Error al tomar foto: ' + result.error) : Alert.alert('Error', result.error);
        }
      });
    };

    var handleGetGPS = function() {
      s.setGpsLoading(true);
      requestPermission().then(function() {
        getCurrentLocation().then(function(result) {
          s.setGpsLoading(false);
          if (result.error) {
            Platform.OS === 'web' ? window.alert('Error GPS: ' + result.error) : Alert.alert('Error GPS', result.error);
            return;
          }
          s.setLatitude(result.latitude);
          s.setLongitude(result.longitude);
        });
      });
    };

    var handleSave = function() {
      var vehiculosData = [];
      for (var i = 0; i < s.vehicleCount; i++) {
        if (!s.vehicles[i].propietario.trim()) {
          Platform.OS === 'web' ? window.alert('El nombre del propietario del vehículo ' + (i + 1) + ' es obligatorio') : Alert.alert('Campo requerido', 'El nombre del propietario del vehículo ' + (i + 1) + ' es obligatorio');
          return;
        }
        vehiculosData.push(Object.assign({}, s.vehicles[i]));
      }
      var now = new Date();
      var fechaStr = now.toLocaleDateString('es-AR') + ' ' + now.toLocaleTimeString('es-AR', { hour: '2-digit', minute: '2-digit' });
      s.setSaving(true);
      insertAccident({
        vehiculos_count: s.vehicleCount,
        vehiculos_json: JSON.stringify(vehiculosData),
        policia: s.policia.trim(),
        movil: s.movil.trim(),
        ambulancia: s.ambulancia.trim(),
        guardia: s.guardia.trim(),
        observaciones: s.observaciones.trim(),
        fecha: fechaStr,
        latitude: s.latitude ? String(s.latitude) : '',
        longitude: s.longitude ? String(s.longitude) : '',
        photo_uri: photo ? photo.uri : ''
      }).then(function() {
        s.setSaving(false);
        Platform.OS === 'web' ? window.alert('Accidente guardado correctamente') : Alert.alert('✅ Guardado', 'Accidente registrado correctamente');
        navigation.goBack();
      }).catch(function(err) {
        s.setSaving(false);
        Platform.OS === 'web' ? window.alert('Error al guardar: ' + (err.message || '')) : Alert.alert('Error', err.message || 'Error al guardar');
      });
    };

    var scrollH = Dimensions.get('window').height - HEADER_HEIGHT - insets.top;

    return React.createElement(View, { testID: 'View-15', style: { flex: 1, backgroundColor: theme.colors.background }, componentId: 'new-accident-screen' },
      React.createElement(View, { testID: 'View-16', style: {
          height: HEADER_HEIGHT + insets.top,
          paddingTop: insets.top,
          backgroundColor: theme.colors.primary,
          flexDirection: 'row',
          alignItems: 'center',
          paddingHorizontal: 16
        },
        componentId: 'new-accident-header'
      },
        React.createElement(TouchableOpacity, { testID: 'TouchableOpacity-3', onPress: function() { navigation.goBack(); }, style: { padding: 8, marginRight: 8 }, componentId: 'btn-back' },
          React.createElement(MaterialIcons, { testID: 'MaterialIcons-6', name: 'arrow-back', size: 26, color: '#FFFFFF' })
        ),
        React.createElement(MaterialIcons, { testID: 'MaterialIcons-7', name: 'add-road', size: 24, color: '#FFFFFF' }),
        React.createElement(Text, { testID: 'Text-18', style: { color: '#FFFFFF', fontSize: 18, fontWeight: 'bold', marginLeft: 8, flex: 1 } }, 'Nuevo Accidente')
      ),
      React.createElement(ScrollView, { testID: 'ScrollView-2', style: Platform.OS === 'web' ? { height: scrollH, overflow: 'auto' } : { flex: 1 },
        contentContainerStyle: { padding: 16, paddingBottom: insets.bottom + SCROLL_EXTRA_PADDING + 16 },
        keyboardShouldPersistTaps: 'handled'
      },
        React.createElement(View, { testID: 'View-17', style: { backgroundColor: theme.colors.card, borderRadius: 12, padding: 16, marginBottom: 16 }, componentId: 'vehicle-count-section' },
          React.createElement(View, { testID: 'View-18', style: { flexDirection: 'row', alignItems: 'center', marginBottom: 14 } },
            React.createElement(MaterialIcons, { testID: 'MaterialIcons-8', name: 'directions-car', size: 20, color: theme.colors.primary }),
